# Sprint 1 — PDF Reader + AI Chat Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Kullanıcı uygulamada PDF açıyor → paragraf seçiyor → "Bu ne demek?" / "Pratik örnek" / "Özet" butonuna basıyor → AI Gemini Flash'tan akan cevap geliyor, kaynak (Eren s.412) linkleniyor, ilgili PDF sayfasına tıklayarak gidebiliyor. Finals (30 Haz) için en kritik özellik.

**Architecture:**
```
PDF Reader (react-pdf) → text selection → ChatPanel
                                          ↓ POST /ai/chat (SSE stream)
Worker /ai/chat:
  1. Embed query via OpenAI text-embedding-3-small
  2. Query Vectorize → top 5 chunks (cosine similarity)
  3. Build RAG prompt: [system + retrieved chunks + selected_text + user question]
  4. Stream Gemini Flash response (SSE chunks)
  5. Persist to D1 chat_history
```

**Tech Stack:**
- Frontend: react-pdf 10 (PDF.js wrapper), Server-Sent Events client (native EventSource)
- Worker: @cloudflare/workers-types Vectorize binding, OpenAI fetch, Gemini streaming fetch
- AI Provider abstraction: TypeScript interface for swappable providers (Gemini default, Anthropic later)

`scripts/ingest-vectorize.ts`:

```ts
import "dotenv/config";
import { readFileSync, readdirSync } from "node:fs";
import { join } from "node:path";

const ACCOUNT_ID = process.env.CLOUDFLARE_ACCOUNT_ID!;
const API_TOKEN = process.env.CLOUDFLARE_API_TOKEN!;  // wrangler login token works via cfetch
const INDEX = "hukuk-vectors";
const EMBED_DIR = "../data/embeddings";

type Chunk = {
  id: string;
  course: string;
  pdf: string;
  page_start: number;
  page_end: number;
  text: string;
  vector: number[];
};

async function upsertBatch(vectors: Array<{ id: string; values: number[]; metadata: any }>) {
  // Vectorize REST API: POST /accounts/{aid}/vectorize/v2/indexes/{name}/upsert
  // Body is NDJSON
  const body = vectors.map((v) => JSON.stringify(v)).join("\n");
  const r = await fetch(
    `https://api.cloudflare.com/client/v4/accounts/${ACCOUNT_ID}/vectorize/v2/indexes/${INDEX}/upsert`,
    {
      method: "POST",
      headers: { Authorization: `Bearer ${API_TOKEN}`, "Content-Type": "application/x-ndjson" },
      body,
    }
  );
  if (!r.ok) throw new Error(`upsert failed ${r.status}: ${await r.text()}`);
}

async function main() {
  const files = readdirSync(EMBED_DIR).filter((f) => f.endsWith(".jsonl"));
  let total = 0;
  for (const f of files) {
    const lines = readFileSync(join(EMBED_DIR, f), "utf8").trim().split("\n");
    console.log(`▶ ${f}: ${lines.length} vectors`);
    const BATCH = 100;
    for (let i = 0; i < lines.length; i += BATCH) {
      const batch = lines.slice(i, i + BATCH).map((line) => {
        const c: Chunk = JSON.parse(line);
        return {
          id: c.id,
          values: c.vector,
          metadata: {
            course: c.course,
            pdf: c.pdf,
            page_start: c.page_start,
            page_end: c.page_end,
            text: c.text.slice(0, 1500), // metadata size limit
          },
        };
      });
      await upsertBatch(batch);
      total += batch.length;
      process.stdout.write(`  ${total} upserted\r`);
    }
    console.log();
  }
  console.log(`Total: ${total} vectors`);
}

main().catch((e) => { console.error(e); process.exit(1); });
```

```ts
export type RetrievedChunk = {
  text: string;
  pdf: string;
  page_start: number;
  page_end: number;
  score: number;
};

export async function embedQuery(query: string, openaiKey: string): Promise<number[]> {
  const r = await fetch("https://api.openai.com/v1/embeddings", {
    method: "POST",
    headers: { Authorization: `Bearer ${openaiKey}`, "Content-Type": "application/json" },
    body: JSON.stringify({ model: "text-embedding-3-small", input: query }),
  });
  if (!r.ok) throw new Error(`embed failed: ${r.status}`);
  const data = await r.json<{ data: Array<{ embedding: number[] }> }>();
  return data.data[0].embedding;
}

export async function retrieve(
  vectorize: VectorizeIndex,
  queryVector: number[],
  course?: string,
  topK = 5
): Promise<RetrievedChunk[]> {
  const filter = course ? { course: { $eq: course } } : undefined;
  const r = await vectorize.query(queryVector, { topK, returnMetadata: "all", filter });
  return r.matches.map((m) => ({
    text: (m.metadata as any).text,
    pdf: (m.metadata as any).pdf,
    page_start: (m.metadata as any).page_start,
    page_end: (m.metadata as any).page_end,
    score: m.score,
  }));
}

export function buildPrompt(
  question: string,
  selectedText: string | undefined,
  chunks: RetrievedChunk[]
): string {
  const context = chunks
    .map((c, i) => `[${i + 1}] ${c.pdf} (s.${c.page_start}-${c.page_end}):\n${c.text}`)
    .join("\n\n");

  return `Sen bir hukuk asistanısın. Aşağıdaki kaynaklardan yararlanarak Türkçe hukuk diliyle, net ve öz cevap ver. Cevabında [1], [2] gibi numaralarla kaynak göster.

KAYNAKLAR:
${context}

${selectedText ? `KULLANICININ SEÇTİĞİ METİN:\n"""${selectedText}"""\n\n` : ""}
KULLANICININ SORUSU:
${question}

CEVAP:`;
}
```

- [ ] **Step 5: Create Gemini provider**

`worker/src/lib/gemini-provider.ts`:
```ts
import type { AIProvider } from "./ai-provider";

export class GeminiProvider implements AIProvider {
  constructor(private apiKey: string, private model = "gemini-2.5-flash") {}

  async *streamChat(prompt: string): AsyncGenerator<string> {
    const r = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/${this.model}:streamGenerateContent?alt=sse&key=${this.apiKey}`,
      {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          contents: [{ parts: [{ text: prompt }] }],
          generationConfig: { temperature: 0.3, maxOutputTokens: 2048 },
        }),
      }
    );
    if (!r.ok || !r.body) throw new Error(`gemini ${r.status}: ${await r.text()}`);

    const reader = r.body.getReader();
    const decoder = new TextDecoder();
    let buffer = "";
    while (true) {
      const { value, done } = await reader.read();
      if (done) break;
      buffer += decoder.decode(value, { stream: true });
      const lines = buffer.split("\n");
      buffer = lines.pop() ?? "";
      for (const line of lines) {
        if (!line.startsWith("data: ")) continue;
        const payload = line.slice(6).trim();
        if (!payload || payload === "[DONE]") continue;
        try {
          const j = JSON.parse(payload);
          const text = j.candidates?.[0]?.content?.parts?.[0]?.text;
          if (text) yield text;
        } catch {
          // partial, ignore
        }
      }
    }
  }
}
```

- [ ] **Step 6: Create AI route**

`worker/src/routes/ai.ts`:
```ts
import { Hono } from "hono";
import { embedQuery, retrieve, buildPrompt } from "../lib/rag";
import { GeminiProvider } from "../lib/gemini-provider";

type Bindings = {
  VECTORIZE: VectorizeIndex;
  OPENAI_KEY: string;
  GEMINI_KEY: string;
  DB?: D1Database;
};

export const ai = new Hono<{ Bindings: Bindings }>();

ai.post("/chat", async (c) => {
  const body = await c.req.json<{
    question: string;
    selected_text?: string;
    course?: string;
    pdf_key?: string;
  }>();

  if (!body.question) return c.json({ error: "question required" }, 400);

  // RAG
  const qVec = await embedQuery(
    body.selected_text ? `${body.selected_text}\n\n${body.question}` : body.question,
    c.env.OPENAI_KEY
  );
  const chunks = await retrieve(c.env.VECTORIZE, qVec, body.course, 5);
  const prompt = buildPrompt(body.question, body.selected_text, chunks);

  // Stream Gemini → SSE
  const provider = new GeminiProvider(c.env.GEMINI_KEY);
  const encoder = new TextEncoder();
  let fullAnswer = "";

  const stream = new ReadableStream({
    async start(controller) {
      // First send sources
      const sources = chunks.map((c) => ({
        pdf: c.pdf, page_start: c.page_start, page_end: c.page_end, score: c.score,
      }));
      controller.enqueue(encoder.encode(`event: sources\ndata: ${JSON.stringify(sources)}\n\n`));

      try {
        for await (const tok of provider.streamChat(prompt)) {
          fullAnswer += tok;
          controller.enqueue(encoder.encode(`event: token\ndata: ${JSON.stringify(tok)}\n\n`));
        }
        controller.enqueue(encoder.encode(`event: done\ndata: ok\n\n`));
      } catch (e) {
        controller.enqueue(encoder.encode(`event: error\ndata: ${JSON.stringify(String(e))}\n\n`));
      }

      // Persist to D1
      if (c.env.DB) {
        await c.env.DB.prepare(
          "INSERT INTO chat_history (id, course, pdf_key, selected_text, question, answer, sources, created_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?)"
        )
          .bind(
            crypto.randomUUID(),
            body.course ?? null,
            body.pdf_key ?? null,
            body.selected_text ?? null,
            body.question,
            fullAnswer,
            JSON.stringify(sources),
            Date.now()
          )
          .run();
      }
      controller.close();
    },
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache",
      Connection: "keep-alive",
    },
  });
});
```

  initialPage?: number;

export function PdfViewer({ url, initialPage = 1, onSelection }: Props) {
  const [pages, setPages] = useState(0);
  const [page, setPage] = useState(initialPage);
  const containerRef = useRef<HTMLDivElement>(null);

    const sel = window.getSelection();
    const text = sel?.toString().trim();
    if (text && text.length > 5 && onSelection) onSelection(text);

  return (

      <div className="flex items-center justify-between p-2 border-b bg-muted/30">
        <button onClick={() => setPage((p) => Math.max(1, p - 1))} className="px-3 py-1 rounded bg-background">‹</button>
        <span className="text-sm">Sayfa {page} / {pages || "?"}</span>
        <button onClick={() => setPage((p) => Math.min(pages, p + 1))} className="px-3 py-1 rounded bg-background">›</button>
      </div>
      <div ref={containerRef} onMouseUp={handleMouseUp} onTouchEnd={handleMouseUp} className="flex-1 overflow-auto bg-muted/20 p-4 flex justify-center">
        <Document file={url} onLoadSuccess={({ numPages }) => setPages(numPages)} loading={<p>PDF yükleniyor...</p>}>
          <Page pageNumber={page} width={Math.min(typeof window !== "undefined" ? window.innerWidth - 32 : 800, 800)} />
        </Document>
      </div>

`frontend/lib/api.ts` sonuna ekle:
```ts
export type ChatSource = { pdf: string; page_start: number; page_end: number; score: number };

export async function* streamChat(params: {
  question: string;
  selected_text?: string;
  course?: string;
  pdf_key?: string;
}): AsyncGenerator<{ type: "sources"; data: ChatSource[] } | { type: "token"; data: string } | { type: "error"; data: string }> {
  const r = await fetch(`${WORKER_URL}/ai/chat`, {
    method: "POST",
    credentials: "include",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(params),
  });
  if (!r.ok || !r.body) throw new Error(`AI chat failed: ${r.status}`);
  const reader = r.body.getReader();
  const decoder = new TextDecoder();
  let buffer = "";
  while (true) {
    const { value, done } = await reader.read();
    if (done) break;
    buffer += decoder.decode(value, { stream: true });
    const events = buffer.split("\n\n");
    buffer = events.pop() ?? "";
    for (const ev of events) {
      const lines = ev.split("\n");
      const event = lines.find((l) => l.startsWith("event: "))?.slice(7);
      const dataLine = lines.find((l) => l.startsWith("data: "))?.slice(6);
      if (!event || !dataLine) continue;
      try {
        const data = JSON.parse(dataLine);
        if (event === "sources") yield { type: "sources", data };
        else if (event === "token") yield { type: "token", data };
        else if (event === "error") yield { type: "error", data };
      } catch {}
    }
  }
}
```

`frontend/components/ChatPanel.tsx`:
```tsx
"use client";
import { useState, useRef, useEffect } from "react";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Card } from "@/components/ui/card";
import { streamChat, type ChatSource } from "@/lib/api";
import { SourceCitation } from "./SourceCitation";

type Msg = { role: "user" | "ai"; content: string; sources?: ChatSource[] };

export function ChatPanel({ selectedText, course, pdfKey }: { selectedText: string | null; course: string; pdfKey: string }) {
  const [messages, setMessages] = useState<Msg[]>([]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const scrollRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    scrollRef.current?.scrollTo({ top: scrollRef.current.scrollHeight, behavior: "smooth" });
  }, [messages]);

  async function send(question: string) {
    if (!question.trim() || loading) return;
    setLoading(true);
    setMessages((m) => [...m, { role: "user", content: question }, { role: "ai", content: "" }]);
    setInput("");
    try {
      for await (const ev of streamChat({ question, selected_text: selectedText ?? undefined, course, pdf_key: pdfKey })) {
        if (ev.type === "sources") {
          setMessages((m) => { const a = [...m]; a[a.length - 1] = { ...a[a.length - 1], sources: ev.data }; return a; });
        } else if (ev.type === "token") {
          setMessages((m) => { const a = [...m]; a[a.length - 1] = { ...a[a.length - 1], content: a[a.length - 1].content + ev.data }; return a; });
        } else if (ev.type === "error") {
          setMessages((m) => { const a = [...m]; a[a.length - 1] = { role: "ai", content: `Hata: ${ev.data}` }; return a; });
        }
      }
    } finally {
      setLoading(false);
    }
  }

  return (
    <Card className="h-full flex flex-col rounded-none md:rounded-l-lg border-l">
      <div ref={scrollRef} className="flex-1 overflow-auto p-3 space-y-3 text-sm">
        {messages.length === 0 && (
          <div className="text-muted-foreground text-center py-8">
            PDF'den paragraf seç + soru sor, ya da aşağıdan doğrudan yaz.
          </div>
        )}
        {messages.map((m, i) => (
          <div key={i} className={m.role === "user" ? "text-right" : ""}>
            <div className={`inline-block max-w-[90%] px-3 py-2 rounded-lg ${m.role === "user" ? "bg-primary text-primary-foreground" : "bg-muted"}`}>
              <div className="whitespace-pre-wrap">{m.content || (m.role === "ai" && loading ? "..." : "")}</div>
              {m.sources && m.sources.length > 0 && (
                <div className="mt-2 flex flex-wrap gap-1">
                  {m.sources.map((s, idx) => <SourceCitation key={idx} src={s} index={idx} />)}
                </div>
              )}
            </div>
          </div>
        ))}
      </div>
      {selectedText && (
        <div className="px-3 py-2 border-t bg-yellow-500/10 text-xs">
          Seçili: "{selectedText.slice(0, 80)}{selectedText.length > 80 ? "..." : ""}"
        </div>
      )}
      <div className="p-2 border-t flex gap-2">
        <Input value={input} onChange={(e) => setInput(e.target.value)} onKeyDown={(e) => e.key === "Enter" && send(input)} placeholder="Sor..." disabled={loading} />
        <Button onClick={() => send(input)} disabled={loading || !input.trim()}>Sor</Button>
      </div>
      {selectedText && (
        <div className="p-2 border-t flex gap-1 flex-wrap">
          <Button size="sm" variant="outline" onClick={() => send("Bu metin ne demek? Sade Türkçe açıkla.")}>Anlat</Button>
          <Button size="sm" variant="outline" onClick={() => send("Bu konuda bir pratik örnek olay ver.")}>Örnek</Button>
          <Button size="sm" variant="outline" onClick={() => send("3 bullet özetle.")}>Özet</Button>
        </div>
      )}
    </Card>
  );
}
```

"use client";
import { useEffect, useState } from "react";
import Link from "next/link";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { COURSES } from "@/lib/courses";
import { api, type PdfListItem } from "@/lib/api";

export default function ReaderLanding() {
  const [byCourse, setByCourse] = useState<Record<string, PdfListItem[]>>({});

  useEffect(() => {
    api.listPdfs("dersler/").then((r) => {
      const grouped: Record<string, PdfListItem[]> = {};
      r.objects.forEach((o) => {
        const parts = o.key.split("/");
        const course = parts[1];
        (grouped[course] ??= []).push(o);
      });
      setByCourse(grouped);
    });
  }, []);

  return (
    <main className="p-6 max-w-4xl mx-auto space-y-4">
      <h1 className="text-2xl font-bold">Dersler</h1>
      {COURSES.map((c) => (
        <Card key={c.id}>
          <CardHeader><CardTitle>{c.name}</CardTitle></CardHeader>
          <CardContent className="space-y-1">
            {(byCourse[c.id] ?? []).map((pdf) => (
              <Link key={pdf.key} href={`/reader/${pdf.key}`} className="block py-1 text-sm hover:bg-muted px-2 rounded">
                {pdf.key.split("/").pop()} <span className="text-muted-foreground">({(pdf.size / 1e6).toFixed(1)} MB)</span>
              </Link>
            ))}
            {(byCourse[c.id]?.length ?? 0) === 0 && (
              <p className="text-sm text-muted-foreground">Bu derste PDF yok.</p>
            )}
          </CardContent>
        </Card>
      ))}
    </main>
  );
}
